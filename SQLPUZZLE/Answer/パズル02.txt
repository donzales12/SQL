-- RULE 1
DELETE FROM Absenteeism
WHERE severity_points >= 40

-- RULE 2
UPDATE Absenteeism
SET    (severity_point = severity_point + 1)
WHERE NOT (
    EXISTS (
        SELECT absent_date
        FROM Absenteeism
        WHERE emp_id = 1
        AND    absent_date = CURRENT_DATE()-1
    ) AND EXISTS (
        SELECT absent_date
        FROM Absenteeism
        WHERE emp_id = 1
        AND    absent_date = CURRENT_DATE()-2
    )
 ) AND(
    WHERE emp_id = 1
 );
